void build(int l,int r,int SeNumber){  
//l,r代表了区间的左右位置，这里其左右位置同时也是我们要处理的区间的位次
    if(l == r){
        d[SeNumber] = a[l];
        return;
    }
    int m = (l+r)/2;
    build(l,m,SeNumber*2);
    build(m+1,r,SeNumber*2+1);
    d[SeNumber] = d[SeNumber*2]+d[SeNumber*2+1];    
}

区间查询

如果要查[3,5]，需要拆分[3,3]&[4,5],合并就可以得到答案，

代码

int getsum(int l, int r, int s, int t, int p) {
  // [l,r] 为查询区间,[s,t] 为当前节点包含的区间,p 为当前节点的编号
  if (l <= s && t <= r)
    return d[p];  // 当前区间为询问区间的子集时直接返回当前区间的和
  int m = (s + t) / 2, sum = 0;
  if (l <= m) sum += getsum(l, r, s, m, p * 2);
  // 如果左儿子代表的区间 [l,m] 与询问区间有交集,则递归查询左儿子
  if (r > m) sum += getsum(l, r, m + 1, t, p * 2 + 1);
  // 如果右儿子代表的区间 [m+1,r] 与询问区间有交集,则递归查询右儿子
  return sum;
}

说实话这个过程对我来说最开始可能是比较直观的，但我发现如果想要在语义层面描述是一件比较困难的事情，现在我尝试一下描述这个过程

首先定义概念，查询区间，查询区间表示起始查询的区间与递归的某个阶段（这个概念我可能用的不是很准确）查询的区间。范围区间表示当前节点所表征的区间（可能学术界有定义，但是我不知道，就只好自己先定义一个了）

递归结束于所有的范围区间包含查询区间

那么从第一 个节点开始进行查询，这个节点包含区间是全集，查询区间必然被包含在这个区间之内（否则信息不全，无法得到结果），然而我们需要使我们所有的范围区间等于查询区间，所以我们对范围区间进行拆分，也就是向下查找其子节点，以期其能够在某一个拆分阶段，以范围区间不多于查询区间的形式进行累加（可加和性的体现，同时也是被包含的原因）

那么就可以看到代码里的两个部分：拆分与包含判断

第一部分是包含判断，也是边界判断，

第二部分是进行拆分，拆分的方式是对范围区间进行拆分，拆分的按照左右节点的规定性进行，也即建树时的二分方式，同时传递节点编号，其实拆分这一步只是建树这一部分的重复

这就是康德所谓的分析判断，哲学计算机不分家

最后是区间修改

区间修改需要所有节点都遍历一次，那还要线段树做什么，时间复杂度直接起飞

所以引入lazytag

即区间修改仍然是针对区间的修改，而不是针对每一个节点的遍历修改

而父节点恰好具有区间的性质…

修改用标记父节点代替修改每一个叶节点，查询时下传lazytag就可以得到相应的值

void update(int l, int r, int c, int s, int t, int p) {
  if (l <= s && t <= r) {
    d[p] = (t - s + 1) * c, b[p] = c;
    return;
  }
  int m = (s + t) / 2;
  if (b[p]) {
    d[p * 2] = b[p] * (m - s + 1), d[p * 2 + 1] = b[p] * (t - m),
          b[p * 2] = b[p * 2 + 1] = b[p];
    b[p] = 0;
  }
  if (l <= m) update(l, r, c, s, m, p * 2);
  if (r > m) update(l, r, c, m + 1, t, p * 2 + 1);
  d[p] = d[p * 2] + d[p * 2 + 1];
}

区间求和

int getsum(int l, int r, int s, int t, int p) {
  // [l,r] 为查询区间,[s,t] 为当前节点包含的区间,p为当前节点的编号
  if (l <= s && t <= r) return d[p];
  // 当前区间为询问区间的子集时直接返回当前区间的和
  int m = (s + t) / 2;
  if (b[p]) {
    // 如果当前节点的懒标记非空,则更新当前节点两个子节点的值和懒标记值
    d[p * 2] += b[p] * (m - s + 1), d[p * 2 + 1] += b[p] * (t - m),
        b[p * 2] += b[p], b[p * 2 + 1] += b[p];  // 将标记下传给子节点
    b[p] = 0;                                    // 清空当前节点的标记
  }
  int sum = 0;
  if (l <= m) sum = getsum(l, r, s, m, p * 2);
  if (r > m) sum += getsum(l, r, m + 1, t, p * 2 + 1);
  return sum;
}

T1 张煊的金箍棒（2）

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>

using namespace std;

int TreeValue[400005]={0};
int lazytag[400005] = {0};

void BuildTree(int from,int to,int SerialNode){
    
    if(from == to){
        TreeValue[SerialNode] = 1;
        return ;    
    }
    int mid = (from + to) / 2;
    BuildTree(from,mid,SerialNode*2);
    BuildTree(from,mid+1,SerialNode*2+1);
    TreeValue[SerialNode] = TreeValue[SerialNode*2]+TreeValue[SerialNode*2+1];
}

void ChangeTree(int ChangeL,int ChangeR,int TraverseL,int TraverseR,int SerialNode,int ChangeValue){
    if(ChangeL<=TraverseL&&TraverseR<=ChangeR){
        TreeValue[SerialNode] = (TraverseR-TraverseL+1)*ChangeValue;
        lazytag[SerialNode] = ChangeValue; 
        return ;
    }
    int mid = (TraverseL + TraverseR)/2;
    if(lazytag[SerialNode]){
        TreeValue[SerialNode*2] = lazytag[SerialNode]*(mid-TraverseL+1);
        TreeValue[SerialNode*2+1] = lazytag[SerialNode]*(TraverseR-mid);
        lazytag[SerialNode*2] = lazytag[SerialNode*2+1] = lazytag[SerialNode];
        lazytag[SerialNode] = 0;
    }
    if(mid >= ChangeL)ChangeTree(ChangeL,ChangeR,TraverseL,mid,SerialNode*2,ChangeValue);
    if(mid <  ChangeR)ChangeTree(ChangeL,ChangeR,mid+1,TraverseR,SerialNode*2+1,ChangeValue);
    TreeValue[SerialNode] = TreeValue[SerialNode*2] + TreeValue[SerialNode*2+1];
}


int SearchTree(int l, int r, int s, int t, int node) {
  if(l<=s&&t<=r)return TreeValue[node];
  int mid = (s+t)/2;
  if(lazytag[node]){
      TreeValue[node*2]+=lazytag[node]*(mid-s+1);
      TreeValue[node*2+1]+=lazytag[node]*(t-mid);
      lazytag[node*2] += lazytag[node];
      lazytag[node*2+1] += lazytag[node];
      lazytag[node] = 0;
  }
  
  int sum = 0;
  if(l<=mid)sum = SearchTree(l,r,s,mid,node*2);
  if(r>mid) sum+= SearchTree(l,r,mid+1,t,node*2+1);
  return sum;
}

int main(){
    int t;
    cin>>t;
    while(t--){
        memset(TreeValue,0,sizeof(TreeValue));
        memset(lazytag,0,sizeof(lazytag));

        int n,q;
        cin>>n;
        cin>>q;
    
        
        BuildTree(1,n,1);
        int x,y,z;
        
        for(int i = 1;i<=q;i++){
            cin>>x>>y>>z;
            ChangeTree(1,n,x,y,1,z);
        }
        cout<<SearchTree(1,n,1,n,1)<<endl;
    }

    return 0;
}

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>

using namespace std;

void push(int p,int l,int r)
{
    ll[p]=ll[ls(p)];
    rr[p]=rr[rs(p)];
    a[p]=max(a[ls(p)],a[rs(p)]);
    int mid=(l+r)>>1;
    if(sum[mid]<sum[mid+1])
    {
        if(ll[ls(p)]==mid-l+1)   ll[p]+=ll[rs(p)];
        if(rr[rs(p)]==r-mid)  rr[p]+=rr[ls(p)];
        a[p]=max(a[p],rr[ls(p)]+ll[rs(p)]);
    }
}
void update(int x,int c,int l,int r,int p)
{
    if(l==r)
    {
        sum[x]=c;
        return ;
    }
    int mid=(l+r)>>1;
    if(x<=mid)  update(x,c,l,mid,ls(p));
    else  update(x,c,mid+1,r,rs(p));
    push(p,l,r);
}
int ask(int x,int y,int l,int r,int p)
{
    if(l>=x&&r<=y)  
        return  a[p];
    int mid=(l+r)/2,ans=-1e9;
    if(x<=mid)      
        ans=max(ans,ask(x,y,l,mid,ls(p)));
    if(y>mid)  
        ans=max(ans,ask(x,y,mid+1,r,rs(p)));
    if(sum[mid]<sum[mid+1]&&x<=mid&&y>mid)              ans=max(ans,min(mid-x+1,rr[ls(p)])+min(y-mid,ll[rs(p)]));
    return ans;
}
void build(int p,int l,int r)
{
    if(l==r)
    {
        ll[p]=rr[p]=a[p]=1;
        return;
    }
    int mid=(l+r)>>1;
    build(ls(p),l,mid);
    build(rs(p),mid+1,r);
    push(p,l,r);
}
